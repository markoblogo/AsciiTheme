Fast setup

Responsive by default

No media overrides

Example code

import "@markoblogo/ascii-theme/base.css";
import { initAsciiTheme } from "@markoblogo/ascii-theme";

initAsciiTheme({
  base: true,
  managedMode: true,
  addThemeToggle: true,
  addStyleToggle: false,
  mountSelector: "#base-toggle-mount"
});
